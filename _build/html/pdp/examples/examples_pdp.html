Define cost

1ess1inv = 5988.87 # ESS investment cost
2c1grid = np.ones((dx02, 1), dtype=float) * 25  # grid cost import
3c2grid = np.ones((dx02, 1), dtype=float) * 25  # grid cost export
4c3grid = np.ones((dx02, 1), dtype=float) * 35  # penality cost

Define cost function

 1def cost3dp(dsoc0, up, dtx0, load2, pv2, sx2, pmax0, ex0, mx0, mx1, ef0, z0, ess0inv,
 2soh0, inv0, c1grid, c2grid, c3grid):
 3    w1 = np.greater(dsoc0, mx0, out=np.ones_like(dsoc0))  # charge
 4    w2 = np.greater(-mx1, dsoc0, out=np.ones_like(dsoc0))  # discharge
 5    w3 = (w1 + w2) * 1e10
 6    m1 = np.greater(dsoc0, 0, out=np.ones_like(dsoc0))  # charging
 7    m2 = np.abs(1 - m1)  # discharge
 8    m3 = (m2 + m1) / ef0
 9    p1bess = -dsoc0 * ex0 / dtx0 * m3
10    netha1 = inv_eff(np.abs(p1bess), inv0)
11    netha2 = inv_in(np.abs(p1bess), inv0)
12    netha2 = inv_eff(netha2, inv0)
13    n6 = np.equal(netha2, 0, out=np.ones_like(netha2))
14    netha2 = np.add(n6, netha2)
15    m4 = np.multiply(m2, netha1) + np.divide(m1, netha2)
16    p0bess = np.multiply(p1bess, m4)  # BESS power
17    n2 = len(p0bess)
18    doh1 = np.multiply(dsoc0, m2) * z0
19    c0ess = -ess0inv * doh1 / (1 - soh0) + w3
20    # <!-- compute_grid_cost -->
21    pv0 = np.tile(pv2[up, :], [n2, 1])
22    l2 = len(pv2[up, :])
23    # print("------")
24    balan4 = -pv0[up, :] + load2[up] # energy balance
25    p1bess = np.tile(np.reshape(p0bess, (n2, 1)), [1, l2])
26    pw0 = balan4 - p1bess  # Grid power [pw0 > 0 :importation]
27    c3gr = np.multiply(pw0, np.greater(0, pw0, out=np.ones_like(pw0))) * c2grid[up]  # export to the grid
28    pw2 = pw0 - pmax0  # excess of energy
29    pw2 = np.multiply(pw2, np.greater(pw2, 0, out=np.ones_like(pw0)))
30    c2gr = pw2 * c3grid[up]  # penalty cost
31    pw1 = pw0 - pw2  # import from grid
32    c1gr = np.multiply(pw1, np.greater(pw1, 0, out=np.ones_like(pw1))) * c1grid[up]
33    # TODO  Compute grid cost
34    c0gr = c1gr + c2gr + c3gr
35    n3 = (np.mean(pv2) == 0) * 1
36    sx3 = np.ones((len(sx2), 1), dtype=float)
37    sx2 = np.add(np.reshape(sx2, (len(sx2), 1)) * (1 - n3), sx3 * n3)
38    # ------------------------- compute_math_expectation_depending_on_pv
39    cpb1 = np.matmul(c0gr, np.flip(sx2))
40    out1 = dict()
41    out1["0"] = np.add(c0ess, np.concatenate(cpb1))  # cost (expectation)
42    out1["1"] = np.concatenate(np.matmul(pw0, sx3) / len(sx3))  # energy from grid
43    return out1
