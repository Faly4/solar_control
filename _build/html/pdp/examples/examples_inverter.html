Compute the efficiency of an inverter rated at 2500W connected to a PV module of 1000Wc

 1import numpy as np
 2from matplotlib import pyplot as plt
 3from pdp import inv_eff
 4
 5inv4rt4pwr = 2500  # Inverter rated power
 6pwr4in = np.linspace(0, 1000, 50)  # Module power output
 7η = inv_eff(pwr4in, inv4rt4pwr)  # Inerter efficiency
 8plt.plot(pwr4in, η)
 9plt.xlabel('Input power [W]')
10plt.title('Inverter efficiency')
11plt.ylabel('η')
12plt.grid()
13plt.show()

Compute the input of an inverter rated at 2500W from its output

 1import numpy as np
 2from matplotlib import pyplot as plt
 3from pdp import inv_in
 4
 5inv4rt4pwr = 2500 # Inverter rated power
 6pwr4out = np.linspace(0,1000,50)  # Inverter power output
 7pwr4in = inv_in(pwr4out, inv4rt4pwr)  # Inverter power input
 8print('Power input', pwr4in)
 9η = np.divide(pwr4out, pwr4in, out=np.zeros_like(pwr4out), where=pwr4in!=0)
10plt.plot(pwr4in, η)
11plt.xlabel('Input power [W]')
12plt.title('Inverter efficiency')
13plt.ylabel('η')
14plt.grid()
15plt.show()
